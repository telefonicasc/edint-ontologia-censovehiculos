
# VEH28: ¿Quiénes son propietarios de vehículos eléctricos con autorización activa?

PREFIX edintveh: <http://vocab.linkeddata.es/datosabiertos/def/transporte/censovehiculos#>
PREFIX edintvehiclepermitstatus: <http://vocab.linkeddata.es/datosabiertos/kos/edint/VehiclePermitStatus/>
PREFIX edintvehiclepropulsiontype: <http://vocab.linkeddata.es/datosabiertos/kos/edint/VehiclePropulsionType/>

SELECT DISTINCT ?owner ?vehicle ?permit
WHERE {
  ?permit a edintveh:Permit ;
          edintveh:authorisesVehicle ?vehicle ;
          edintveh:permitStatus edintvehiclepermitstatus:Active .

  ?vehicle a edintveh:Vehicle ;
           edintveh:registeredOwner ?owner ;
           edintveh:propulsionType edintvehiclepropulsiontype:Electric .
}
ORDER BY ?owner ?vehicle
